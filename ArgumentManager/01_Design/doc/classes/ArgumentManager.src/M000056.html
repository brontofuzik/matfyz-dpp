  def ArgumentManager.expand_short_option_clusters( argv )
    new_argv = []
    argv.each { | x |
      if ArgumentManager.match_short_option_cluster( x )
        # need to expand
        ArgumentManager.expand_short_option_cluster( x ).each { | x |
          new_argv.push( x )
        }
      else
        # no need to expand
        new_argv.push(x)
      end
    }
    return new_argv
  end
