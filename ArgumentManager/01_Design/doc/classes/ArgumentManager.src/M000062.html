  def ArgumentManager.match_option( option )
    # short option
    match_data = ArgumentManager.match_short_option( option )
    if match_data
      return match_data
    end
    # long option
    match_data = ArgumentManager.match_long_option( option )
    if match_data
      return match_data
    end
    # no match (non-option)
    return nil
  end
