  def ArgumentManager.set_defaults( specs = {} )
    specs.each {
      | spec |
      case spec[ 0 ]
        when :type
          ArgumentManager.set_default_type( spec[ 1 ] )
        when :compulsory
          ArgumentManager.set_default_compulsory( spec[ 1 ] )
        when :min
          ArgumentManager.set_default_min( spec[ 1 ] )
        when :max
          ArgumentManager.set_default_max( spec[ 1 ] )
        when :df_int
          ArgumentManager.set_default_df_int( spec[ 1 ] )
        when :domain
          ArgumentManager.set_default_domain( spec[ 1 ] )
        when :df_str
          ArgumentManager.set_default_df_str( spec[ 1 ] )
        when :help
          ArgumentManager.set_default_help( spec[ 1 ] )
        when :bind
          ArgumentManager.set_default_bind( spec[ 1 ] )
        else
          puts 'exception: invalid specification keyword'
      end
    }
  end
