  def ArgumentManager.add_option( names = [], specs = {} )
    # default option type: boolean
    if specs[ :type ] == nil
      specs[ :type ] = 'boolean'
    end
    
    if names.is_a?(Array)
      name.each {
        | x |
        puts "#{ x }, type: #{ specs[ :type ] }"
      }
    else
      puts "#{ name }, type: #{ specs[ :type ] }"
    end
    
    # (Boolean|Integer|String)Option.new( names, specs)
    code = "@@options.push " + specs[ :type ].capitalize + "Option.new( names, specs )"
    eval code
  end
