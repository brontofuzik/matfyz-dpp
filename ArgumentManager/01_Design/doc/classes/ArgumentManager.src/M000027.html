  def ArgumentManager.add_option( names = [], specs = {} )
    # names is converted to an array if necessary
    names = names.to_a
    
    # specification composition (user specified + defaults)
    specs = @@df_specs.merge( specs )
    
    # ( Boolean | Integer | String )Option.new( names, specs )
    code = "@@options.push #{specs[ :type ].capitalize}Option.new( names, specs )"
    eval code
  end
